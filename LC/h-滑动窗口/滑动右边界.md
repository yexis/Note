###### 例题

> [3298. 统计重新排列后包含另一个字符串的子字符串数目 II](https://leetcode.cn/problems/count-substrings-that-can-be-rearranged-to-contain-a-string-ii/)
>
> ```cpp
> class Solution {
> public:
>     using ll = long long;
>     long long validSubstringCount(string s, string t) {
>         int n = s.size(), m = t.size();
>         vector<int> cnt(26);
>         for (auto& ch : t) {
>             cnt[ch - 'a']++;
>         }
> 
>         vector<int> c(26);
>         int cur = 0;
> 
>         ll ans = 0;
>         int l = 0, r = -1;
>         while (l < n) {
>             int ch;
>             while (r < n && cur < m) {
>                 ++r;
>                 if (r >= n) break;
>                 ch = s[r] - 'a';
>                 if (++c[ch] <= cnt[ch]) {
>                     cur++;
>                 }
>             }
>             ans += (n - r);
>             
>             ch = s[l] - 'a';
>             if (--c[ch] < cnt[ch]) {
>                 cur--;
>             }
>             ++l;
>         }
>         return ans;
>     }
> };
> ```
