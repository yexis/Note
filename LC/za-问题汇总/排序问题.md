```cpp
#include <bits/stdc++.h>
using namespace std;

using i64 = long long;

int main() {
    int n;
    cin >> n;

    vector<string> in;
    for (int i = 1; i <= n; i++) {
        string s;
        cin >> s;
        
        for (int i = 0; i < s.size(); i++) {
            if (s[i] > s[i + 1] || i == s.size() - 1) {
                s.erase(i, 1);
                break;
            }
        }
        
        if (s.size()) in.push_back(s);
    }

  	// 在这个排序中，如果存在空串，则会有问题
  	// 假如 s = "A", t = ""
  	// 那么 
    sort(in.begin(), in.end(), [&](auto s, auto t) {
        return s + t < t + s;
    });

    string ans;
    for (auto &s : in) {
        ans += s;
    }
    cout << ans << "\n";
}

```

void solve() {

}
