class Solution {
public:
    int findMaximumXOR(vector<int>& a) {
        // 方法1: 0/1字典树
        
        // 方法2: 试填法
        int n = a.size();
        int max_or = 0; for (int e : a) max_or |= e;
        if (max_or == 0) return 0;

        int m = 32 - __builtin_clz(max_or);
        
        int ans = 0;
        for (int d = m - 1; d >= 0; d--) {
            int nxt = ans | (1 << d);
            unordered_set<int> st;
            for (int i = 0; i < n; i++) {
                int x = a[i] >> d << d;
                int y = x ^ nxt;
                if (st.count(y)) {
                    ans = nxt;
                    break;
                }
                st.insert(x);
            }
            
        }
        return ans;
    }
};
