##### 一 给定一个数x，如何计算x的每一位上有多少个1

例题：
https://ac.nowcoder.com/acm/contest/86391/D

```cpp
void cal(ll x) {
    for (int i = 0; i < 62; i++) {
        ll one = 0;
        ll rc = (x + 1) / (1ll << (i + 1));
        ll rr = (x + 1) % (1ll << (i + 1));
        // 第i位的1的数目
        one += rc * (1ll << i) + max(0ll, rr - (1ll << i));
        cout << i << " " << one << "\n";
    }
    return;
}
```

**方法一：**

```cpp
// 统计第i位的1数量
ll cal(ll x, int i) {
    // 为什么要加1?
    ll c = (x + 1) / (1ll << (i + 1));
    ll r = (x + 1) % (1ll << (i + 1));
    // 第i位的1的数目
    return c * (1ll << i) + max(0ll, r - (1ll << i));
}

int main() {
    ll T;
    cin >> T;
    while (T--) {
        ll l, r;
        cin >> l >> r;
        ll ans = 0;
        for (int i = 0; i < 62; i++) {
            // 计算第i位有几个1
            ll one = 0;
            one += cal(r, i);
            one -= cal(l - 1, i);
            if (one & 1) {
                ans += (1ll << i);
            }
        }
        cout << ans << "\n";
    }
    return 0;
}
```

**方法二：巧妙法**

**结论：**

**$4k，4k+1,4k+2，4k+3$ 这四个数的异或和为0**

**证明：**

```cpp
0: ...0 0 0
1: ...0 0 1
2: ...0 1 0
3: ...0 1 1
-------------以上4项异或和为0
4: ...1 0 0 
5: ...1 0 1
6: ...1 1 0
7: ...1 1 1
-------------以上4项异或和为0
8:  ...1 0 0 0   对4取余为0，前缀异或和为 x
9:  ...1 0 0 1   对4取余为1，前缀异或和为 1
10: ...1 0 1 0   对4取余为2，前缀异或和为 x + 1
11: ...1 0 1 1   对4取余为3，前缀异或和为 0
-------------以上4项异或和也为0
...
```



```cpp
// 写法四
ll cal(ll x) {
    if (x % 4 == 1) {
        return 1;
    } else if (x % 4 == 2) {
        return x + 1;
    } else if (x % 4 == 3) {
        return 0;
    } else {
        return x;
    }
}

int main() {
    ll T;
    cin >> T;
    while (T--) {
        ll l, r;
        cin >> l >> r;
        ll ans = cal(r) ^ cal(l - 1);
        cout << ans << "\n";
    }
    return 0;
}
```
