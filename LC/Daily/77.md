func dfs(index, count, k, n int) {
	if count > k {
		var cp []int
		cp = append(cp, s...)
		res = append(res, cp)
		return
	}
	for i := index; i <= n; i++ {
		s = append(s, i)
		dfs(i + 1, count + 1, k, n)
		s = s[:len(s) - 1]
	}
	return
}
