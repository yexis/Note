[删除重复字符后的字典序最小字符串](https://leetcode.cn/contest/weekly-contest-485/problems/lexicographically-smallest-string-after-deleting-duplicate-characters/description/)

```cpp
class Solution {
public:
    string lexSmallestAfterDeletion(string s) {
        int n = s.size();
        vector<int> cnt(26); for (int i = 0; i < n; i++) cnt[s[i] - 'a']++;

        auto left = cnt; // 剩余未被删除的元素
      
        string ans;
        for (int i = 0; i < n; i++) {
            while (ans.size() && ans.back() > s[i] && left[ans.back() - 'a'] > 1 ) {
                left[ans.back() - 'a']--;
                ans.pop_back();
            } 
            ans += s[i];
        }

        while (ans.size() > 1) {
            if (left[ans.back() - 'a'] > 1) {
                left[ans.back() - 'a']--;
                ans.pop_back();
            } else {
                break;
            }
        }
        
        return ans;
    }
};©leetcode
```
