原地合并

```cpp
    // a b 均为已排序数组
		void merge(vector<int>& a, vector<int>& b) {
        int na = a.size(), nb = b.size();
        a.resize(na + nb);
        int i = na - 1, j = nb - 1, k = na + nb - 1;
        for (; i >= 0 && j >= 0;) {
            if (a[i] >= b[j]) {
                a[k--] = a[i--];
            } else {
                a[k--] = b[j--];
            }
        }
        while (i >= 0) a[k--] = a[i--];
        while (j >= 0) a[k--] = b[j--];
    }
```
