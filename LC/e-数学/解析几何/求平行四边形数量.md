struct KB {
    pii k;
    pii b;
};

// 计算直线(二元一次方程) 斜率k 和 截距b
// 使用pii(1, 0)表示斜率为0，平行于x轴，此时 b = y
// 使用pii(0, 1)表示斜率为正无穷，平行于y轴，此时 b = x
KB cal_kb(int x1, int y1, int x2, int y2) {
    KB kb; 
    auto& k = kb.k; 
    auto& b = kb.b;
    
    int dx = x2 - x1;
    int dy = y2 - y1;

    int g = gcd(dx, dy);
    dx /= g, dy /= g;
    if (dx < 0) dx = -dx, dy = -dy;

    k.first = dx, k.second = dy;

    if (k.first == 0) {
        // k 正无穷
        // k = pii(0, 1)
        k.second = 1;
        b.first = 1, b.second = x1;
    } else if (k.second == 0) {
        // k == 0
        // k = pii(1, 0)
        k.first = 1;
        b.first = 1, b.second = y1;
    } else {
        // b = y - (dy / dx) * x = y * dx - dy * x / dx
        b.first = dx, b.second = y1 * dx - dy * x1;
        g = gcd(b.first, b.second);
        b.first /= g, b.second /= g;
    }

    return kb;
}

// 统计平行四边形的个数
// 性质：平行四边形的两条对角线终点重合
// 所有对于中点重合的任意两条对角线，其一定能组成一个平行四边形
// 注意：如果两条对角线共线，那么无法组成平行四边形，需要排除
int cal_parallelogram(vector<vector<int>>& ps) {
    int n = ps.size();
    int ans = 0;
    // 中点在mid的斜率为k的线段个数
    map<pii, map<pii, int>> V;
  	// 按对角线遍历
    for (int i = 0; i < n; i++) {
        int x1 = ps[i][0], y1 = ps[i][1];
        for (int j = i + 1; j < n; j++) {
            int x2 = ps[j][0], y2 = ps[j][1];
            pii know = cal_kb(x1, y1, x2, y2).k;
            
            int xmid = (x1 + x2), ymid = (y1 + y2);
            pii mid = pii(xmid, ymid);
            
            for (auto& [k, v] : V[mid]) {
                if (k == know) continue;
                ans += v;
            }
            V[mid][know]++;
        }
    }
    return ans;
}
