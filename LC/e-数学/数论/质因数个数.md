> 计算元素x的质因数个数，**每个质因数只计算一次**
>
> 比如：
>
> 3 = 3				  pv[3] = 1
>
> 6 = 2 * 3			pv[6] = 2
>
> 8 = 2 * 2 * 2      pv[8] = 1
>
> 12 = 2 * 2 * 3    pv[12] = 2



##### 素数筛

```cpp
using ll = long long;
const int mod = 1e9 + 7;
bool fg = false;
constexpr static int maxn = 1e5 + 5;
// pv[i]表示i的除1和i外的质因子数量
int pv[maxn];
void init() {
    memset(pv, 0, sizeof(pv));
    for (int i = 2; i < maxn; i++) {
        int x = i, cnt = 0;
        for (int j = 2; j * j <= x; j++) {
            if (x % j == 0) {
                while (x % j == 0) {
                    x /= j;
                }
                cnt++;
            }
        }
        if (x > 1) {
            cnt++;
        }
        pv[i] = cnt;
    }
}
```



##### 线性筛

```cpp
using ll = long long;
const int mod = 1e9 + 7;
bool fg = false;
constexpr static int maxn = 1e5 + 5;
// pv[i]表示i的除1和i外的质因子数量
int pv[maxn];
void init() {
    memset(pv, 0, sizeof(pv));
    for (int i = 2; i < maxn; i++) {
        // i非质数
        if (pv[i]) continue;
        // i是质数
        for (int j = i; j < maxn; j += i) {
            pv[j]++; // i 是 j 的一个质因子
        }
    }
}

```



##### 欧式筛

```cpp
const int N = 1e6;
vector<int> prime;
bool not_prime[N + 1];

void init() {
  	not_prime[0] = not_prime[1] = true;
    for (int i = 2; i <= N; ++i) {
        if (!not_prime[i]) {
            prime.push_back(i);
        }
        for (int p : prime) {
            if (i * p > N) break;
            not_prime[i * p] = true;
            if (i % p == 0) {
                // i % p == 0
                // 换言之，i 之前被 p 筛过了
                // 由于 prime 里面质数是从小到大的，所以 i 乘上其他的质数的结果一定会被 p 的倍数筛掉
                // 就不需要在这里先筛一次，所以这里直接 break 掉就好了
                break;
            }
        }
    }
}
```
