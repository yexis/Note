using ll = long long;
void solve2() {
    ll fac[30];
    memset(fac, 0, sizeof(fac));
    fac[0] = 1;
    for (int i = 1; i <= 21; i++) {
        fac[i] = fac[i - 1] * i;
    }
    for (int i = 0; i <= 21; i++) {
        cout << i << "! = " << fac[i] << "\n";
    }
}
int main() {
    solve2();
    return 0;
}
